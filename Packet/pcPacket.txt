enum Type : uint8
{
	LOGIN_REQUEST = 0
	LOGIN_RESPONSE = 1
	TYPE_NUM = 2
}

packet Header
{
	Type type
}

packet LoginRequest : Header
{
	int8 idLength
	int8 passwordLength
	string16 id
	string16 password
}

enum LoginResult : uint8
{
	FAILED = 0
	SUCCESS = 1
}

packet LoginResponse : Header
{
	LoginResult result
}
