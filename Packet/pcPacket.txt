enum Type : uint8
{
	LOGIN_REQUEST = 0
	LOGIN_RESPONSE = 1
    ALLOC_HERO = 2
	TYPE_NUM = 3
}

enum HeroClass : uint8
{
    FIGHTER = 0
    MAGICIAN = 1
    ARCHER = 2
    THIEF = 3
    PRIEST = 4
    MONK = 5
}

packet Header
{
	Type type
}

packet LoginRequest : Header
{
	int8 idLength
	int8 passwordLength
	string16 id
	string16 password
}

enum LoginResult : uint8
{
	FAILED = 0
	SUCCESS = 1
}

packet LoginResponse : Header
{
	LoginResult result
}

packet AllocHero : Header
{
    int8 allocNum
    HeroClass heroClass arr 4
    int8 x arr 4
    int8 y arr 4
}
